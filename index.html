        function handleWordClick(element, e) {

            e.stopPropagation();

            

            const wordText = element.getAttribute('data-word');

            const wordId = element.getAttribute('data-id');

            const entry = dictionary[wordId];

            

            if (!entry) return;


            // Als we op hetzelfde woord klikken...

            if (activeWordData && activeWordData.word === wordText) {

                if (!activeWordData.showTranslation) {

                    // Tweede klik: Toon vertaling

                    activeWordData.showTranslation = true;

                    renderTooltipContent();

                } else {

                    // Derde klik: Sluit tooltip

                    closeTooltip();

                }

                return;

            }


            // Nieuw woord aangeklikt

            closeTooltip(); // Sluit eventueel vorig woord

            activeWordElement = element;

            activeWordElement.classList.add('word-active');

            

            activeWordData = {

                word: wordText,

                ...entry,

                showTranslation: false

            };


            renderTooltipContent();

            ui.tooltip.classList.remove('hidden');

            positionTooltip(element);

        }

        function positionTooltip(targetElement) {

            const rect = targetElement.getBoundingClientRect();

            const containerRect = ui.appContainer.getBoundingClientRect();

            

            const tooltipWidth = 320; 

            const padding = 20;

            const containerWidth = ui.appContainer.offsetWidth;

            const containerHeight = ui.appContainer.offsetHeight;

            

            // X Positie

            let displayLeft = rect.left - containerRect.left + (rect.width / 2);

            const minLeft = tooltipWidth / 2 + padding;

            const maxLeft = containerWidth - (tooltipWidth / 2) - padding;

            const constrainedLeft = Math.max(minLeft, Math.min(maxLeft, displayLeft));

            const arrowOffset = displayLeft - constrainedLeft;


            // Y Positie (van onderaf gemeten voor veilige expansie)

            const topPosRelativeToContainer = rect.top - containerRect.top;

            const bottomPos = containerHeight - topPosRelativeToContainer + 15;


            ui.tooltip.style.bottom = `${bottomPos}px`;

            ui.tooltip.style.left = `${constrainedLeft}px`;

            ui.ttArrow.style.left = `calc(50% + ${arrowOffset}px)`;

        }


        // Event delegatie voor de interactieve woorden en buitenkliks

        document.addEventListener('click', (e) => {

            const wordEl = e.target.closest('.word-interactive');

            if (wordEl) {

                handleWordClick(wordEl, e);

            } else if (!e.target.closest('#tooltip')) {

                closeTooltip();

            }

        });

                utterance.lang = 'en-GB';

                utterance.rate = 0.85;

        function toggleReadFullText(e) {

            e.stopPropagation();

            if (!('speechSynthesis' in window)) return;


            if (isReadingFull) {

                stopSpeech();

                return;

            }


            window.speechSynthesis.cancel();

            

            // Haal platte tekst op zonder HTML (vergelijkbaar met innerText in React ref)

            const textToRead = ui.textContent.innerText || "";

            

            const utterance = new SpeechSynthesisUtterance(textToRead);

            utterance.lang = 'en-GB';

            utterance.rate = 0.85;


            const voice = getPreferredVoice();

            if (voice) utterance.voice = voice;


            utterance.onend = () => { isReadingFull = false; updateReadFullButtonUI(); };

            utterance.onerror = () => { isReadingFull = false; updateReadFullButtonUI(); };


            isReadingFull = true;

            updateReadFullButtonUI();

            window.speechSynthesis.speak(utterance);

        }


        function stopSpeech() {

            if ('speechSynthesis' in window) {

                window.speechSynthesis.cancel();

            }

            isReadingFull = false;

            updateReadFullButtonUI();

        }


        function updateReadFullButtonUI() {

            if (isReadingFull) {

                ui.readFullBtn.className = "flex items-center gap-2 px-5 py-2.5 rounded-full font-black text-[10px] uppercase tracking-widest transition-all bg-red-100 text-red-600 shadow-inner border border-red-200";

                ui.readFullIcon.innerHTML = `<span class="animate-pulse">${icons.VolumeX}</span>`;

                ui.readFullText.innerText = "Stop reading";

            } else {

                ui.readFullBtn.className = "flex items-center gap-2 px-5 py-2.5 rounded-full font-black text-[10px] uppercase tracking-widest transition-all bg-indigo-50 text-indigo-600 hover:bg-indigo-100 hover:shadow-md border border-indigo-100 hover:scale-105";

                ui.readFullIcon.innerHTML = icons.Volume2Small;

                ui.readFullText.innerText = "Read text";

            }

        }


        ui.readFullBtn.addEventListener('click', toggleReadFullText);

            loadVoices();

            if ('speechSynthesis' in window) {

                window.speechSynthesis.onvoiceschanged = loadVoices;

            }

            renderAll();

        };

            if ('speechSynthesis' in window) window.speechSynthesis.cancel();

        });
